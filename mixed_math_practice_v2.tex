\section*{2. First-order ODEs: solutions}
\begin{enumerate}
    \item $y=3\sqrt{1+x^2}$.
    \item $\arctan y=e^x+C$, hence $y=\tan(e^x+C)$.
    \item $y=1+C(x+2)$.
    \item $y^3=x^3+3$, hence $y=(x^3+3)^{1/3}$.
    \item $y=3e^{-x}-e^{-3x}$.
    \item $y=x^3+Cx^2$.
    \item $y=e^{-x}\!\left(\frac{x}{2}(\sin x+\cos x)+\frac{1}{2}\sin x+C\right)$.
    \item $y=x\left(\frac{x}{2}\ln x-\frac{x}{4}+\frac{1}{4}\right)$.
    \item Potential: $F(x,y)=x^2+xy+y^2=C$.
    \item Potential: $F(x,y)=x^2y+\sin x+\cos y=C$.
    \item Potential: $F(x,y)=x^3+2xy+y^4=C$; using $y(0)=1$, $C=1$.
    \item Potential: $F(x,y)=e^x+xy+y^2=C$.
    \item $y=\dfrac{1}{Ce^{2x}+\frac{x}{2}-\frac14}$.
    \item $y=\pm\dfrac{1}{x\sqrt{C-x^4}}$.
    \item $y=\left(\dfrac{2}{3}e^{-x}+Ce^{-x/2}\right)^2$.
    \item $y=\dfrac{1}{Cx^2+x}$.
    \item Since $dy/dx+\frac{1}{x}y=-2$, an integrating factor is $\mu(x)=x$, and a first integral is $xy+x^2=C$.
    \item Here $M=2xy+x$, $N=x^2$. Since $\dfrac{\partial M/\partial y-\partial N/\partial x}{N}=0$, the equation is already exact after simplification as $y'+\frac{2}{x}y=-\frac{1}{x}$ for $x\neq 0$, giving $x^2y+\frac{x^2}{2}=C$.
    \item Writing $y'+\frac{3}{x}y=-\frac{1}{x}$, an integrating factor is $x^3$, so $x^3y+\frac{x^3}{3}=C$. Using $y(1)=0$, $C=\frac13$, hence $y=\dfrac{1-x^3}{3x^3}$.
    \item Since $y'+y=-x-1$, we get $y=-x+Ce^{-x}$. Using $y(0)=1$, $C=1$, hence $y=1-x+e^{-x}-1=e^{-x}-x$.
\end{enumerate}
