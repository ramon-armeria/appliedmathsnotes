\newcommand{\writelines}[1]{%
	\vspace*{#1}
	\hrule
}

% ---------- Full reading sheet ----------
\newcommand{\MathReadingTemplate}{
	\section*{Reading Sheet}
	
	\textbf{Book / Chapter / Section:}
	
	\writelines{0.8cm}
	
	\textbf{Page / Passage:}
	
	\writelines{0.8cm}
	
	\textbf{Topic:}
	
	\writelines{0.8cm}
	
	\textbf{Role:}
	
	\writelines{0.8cm}
	
	\section*{Question}
	\textbf{Hidden question}
	
	\writelines{2cm}
	
	\section*{Content}
	\textbf{Statement / notation / assumptions / conclusion}
	
	\writelines{3cm}
	
	\section*{Structure}
	\textbf{What is the real point?}
	
	\writelines{3cm}
	
	\section*{Place}
	\textbf{Why here? What gap does it fill?}
	
	\writelines{2cm}
	
	\section*{Dependency}
	\textbf{What does it rely on?}
	
	\writelines{2cm}
	
	\section*{Necessity}
	\textbf{What fails without it?}
	
	\writelines{3cm}
	
	\section*{Mechanism}
	\textbf{Proof strategy / construction / key move}
	
	\writelines{2.5cm}
	
	\section*{Pattern}
	\textbf{Logical skeleton}
	
	\writelines{2cm}
	
	\section*{Forward}
	\textbf{What does it prepare?}
	
	\writelines{2cm}
	
	\section*{Tension}
	\textbf{What still feels unresolved?}
	
	\writelines{2.5cm}
	
	\section*{Compression}
	\textbf{One-line version}
	
	\writelines{1.2cm}
	
	\textbf{Three-line version}
	
	\writelines{2cm}
	
	\section*{Rule}
	\begin{enumerate}[leftmargin=2em]
		\item the question
		\item the mathematical answer
		\item the structural role
	\end{enumerate}
	
	\section*{Master prompt}
	\emph{What is this doing, why now, and what breaks without it?}
}
